  const rsvpBtn = document.getElementById('rsvpBtn');

  const rsvpForm = document.getElementById('rsvpForm');


  rsvpBtn.addEventListener('click', () => {

    const expanded = rsvpBtn.getAttribute('aria-expanded') === 'true';

    rsvpBtn.setAttribute('aria-expanded', !expanded);

    rsvpForm.style.display = expanded ? 'none' : 'block';

    rsvpForm.setAttribute('aria-hidden', expanded);

  });


  rsvpForm.addEventListener('submit', (e) => {

    e.preventDefault();

    alert('Grazie per la conferma! Non vediamo l\'ora di vederti.');

    rsvpForm.reset();

    rsvpForm.style.display = 'none';

    rsvpBtn.setAttribute('aria-expanded', 'false');

    rsvpForm.setAttribute('aria-hidden', 'true');

  });
